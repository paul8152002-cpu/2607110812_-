!/bin/bash

 =============================================

1단계: projectA 그룹 만들었어요.

# -----------------------------------------------

# groupadd 명령어로 새로운 그룹을 만들었어요.
# 이 그룹에 속한 사람들만 /projectA 폴더에 접근할 수 있게 만들었어요.
sudo groupadd projectA

# -----------------------------------------------

# 2단계: 신규 팀원 2명 생성 후 그룹에 설정했어요.

# -----------------------------------------------

# useradd로 새 사용자를 만들었으요.
# -m 옵션은 홈 디렉토리(/home/member1 등)를 자동으로 만들어주고,
# -G projectA 옵션은 생성과 동시에 projectA 그룹에 넣어줬어요.
sudo useradd -m -G projectA member1
sudo useradd -m -G projectA member2

# -----------------------------------------------

# 3단계: /projectA 전용 디렉토리 생성.

# -----------------------------------------------

# mkdir로 루트 경로에 projectA 전용 폴더를 만들었어요.
# 이 폴더가 팀의 공유 작업 공간이자 금고 역할을 해요.
sudo mkdir /projectA

# -----------------------------------------------

# 4단계: 디렉토리 소유 그룹을 projectA로 설정.

# -----------------------------------------------

# chown으로 폴더의 소유자와 소유 그룹을 지정했어요.
# root가 소유자이고, projectA 그룹이 이 폴더를 관리해요.
sudo chown root:projectA /projectA

# -----------------------------------------------

# 5단계: 권한 770 설정으로 외부인 완전 차단합니다.

# -----------------------------------------------

# chmod 770의 의미를 숫자로 풀어보면:
#   7 (소유자 root)  → 읽기(4) + 쓰기(2) + 실행(1) = 7 
#   7 (그룹 projectA) → 읽기(4) + 쓰기(2) + 실행(1) = 7 
#   0 (외부인 others) → 아무 권한 없음 = 0 
# 그룹 외의 사람은 절대 접근 불가능합니다.
sudo chmod 770 /projectA

# =============================================

# 검증: id와 ls -l 명령어로 완벽함 증명.

# =============================================

echo "=== 팀원 그룹 소속 확인 ==="
# id 명령어로 각 유저가 projectA 그룹에 잘 들어갔는지 확인합니다.
# 출력 결과에 groups=...(projectA) 가 보이면 성공입닠다!
id member1
id member2

echo ""
echo "=== /projectA 디렉토리 권한 확인 ==="
# ls -ld 로 폴더 자체의 권한 정보를 확인해요.
# 결과가 drwxrwx--- 로 시작하면 외부인 차단이 완벽하게 된 거예요!
ls -ld /projectA
